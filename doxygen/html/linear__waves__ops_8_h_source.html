15{
16 void operator()(
17 LinearWaves::DataType& data, const ::amr_wind::utils::MultiParser& pp)
18 {
19 auto& wdata = data.meta();
20 auto& info = data.info();
21 relaxation_zones::read_inputs(wdata, info, pp);
22 // Get gravity from MultiPhase physics, assume negative z
23 wdata.g = -data.sim().physics_manager().get<MultiPhase>().gravity()[2];
24
25 pp.get("wave_length", wdata.wave_length);
26 pp.get("wave_height", wdata.wave_height);
27 pp.query("wave_phase_offset_radians", wdata.wave_phase_offset);

30 wdata.wave_phase_offset *= M_PI / 180.;
31 } else if (pp.contains("wave_phase_offset_degrees")) {
32 amrex::Abort(
33 "ReadInputsOp<LinearWaves> : wave phase offset is specified in "
34 "both radians and degrees. Please use only one.");
35 }

40struct InitDataOp<LinearWaves>

42 void operator()(
43 LinearWaves::DataType& data, int level, const amrex::Geometry& geom)
44 {
45 const auto& wdata = data.meta();
46
47 auto& sim = data.sim();
48
49 // cppcheck-suppress constVariableReference
50 auto& m_levelset = sim.repo().get_field("levelset");
51 // cppcheck-suppress constVariableReference
52 auto& m_velocity = sim.repo().get_field("velocity");
53
54 const auto& problo = geom.ProbLoArray();
55 const auto& probhi = geom.ProbHiArray();
56 const auto& dx = geom.CellSizeArray();
57
58 const auto& phi = m_levelset(level).arrays();
59 const auto& vel = m_velocity(level).arrays();
60
61 const amrex::Real zero_sea_level = wdata.zsl;
62 const amrex::Real gen_length = wdata.gen_length;
63 const amrex::Real beach_length = wdata.beach_length;
64 const amrex::Real g = wdata.g;
65 const bool has_beach = wdata.has_beach;
66 const bool init_wave_field = wdata.init_wave_field;
67
68 const amrex::Real wave_height = wdata.wave_height;
69 const amrex::Real wave_length = wdata.wave_length;
70 const amrex::Real phase_offset = wdata.wave_phase_offset;
71 const amrex::Real water_depth = wdata.water_depth;
72 const amrex::Real current = wdata.current;
73
74 amrex::ParallelFor(
75 m_velocity(level), amrex::IntVect(3),
76 [=] AMREX_GPU_DEVICE(int nbx, int i, int j, int k) noexcept {
77 const amrex::Real x = problo[0] + (i + 0.5) * dx[0];
78 const amrex::Real z = problo[2] + (k + 0.5) * dx[2];
79
80 const amrex::Real wave_number = 2. * M_PI / wave_length;
81 const amrex::Real omega = std::pow(
82 wave_number * g * std::tanh(wave_number * water_depth),
83 0.5);
84 const amrex::Real phase = wave_number * x - phase_offset;
85
86 // Wave profile
87 const amrex::Real eta_w =
88 wave_height / 2.0 * std::cos(phase) + zero_sea_level;
89 const amrex::Real u_w =
90 omega * wave_height / 2.0 *
91 std::cosh(
92 wave_number * (z - zero_sea_level + water_depth)) /
93 std::sinh(wave_number * water_depth) * std::cos(phase);
94 const amrex::Real v_w = 0.0;
95 const amrex::Real w_w =
96 omega * wave_height / 2.0 *
97 std::sinh(
98 wave_number * (z - zero_sea_level + water_depth)) /
99 std::sinh(wave_number * water_depth) * std::sin(phase);
100 const utils::WaveVec wave_sol{u_w, v_w, w_w, eta_w};
101
102 // Quiescent profile
103 const utils::WaveVec quiescent{0.0, 0.0, 0.0, zero_sea_level};
104
105 // Specify initial state for each region of domain
106 const auto bulk = init_wave_field ? wave_sol : quiescent;
107 const auto outlet = has_beach ? quiescent : wave_sol;
108
109 const auto local_profile = utils::harmonize_profiles_1d(
110 x, problo[0], gen_length, probhi[0], beach_length, wave_sol,
111 bulk, outlet);
112
113 phi[nbx](i, j, k) = local_profile[3] - z;
114 const amrex::Real cell_length_2D =
115 std::sqrt(dx[0] * dx[0] + dx[2] * dx[2]);
116 if (phi[nbx](i, j, k) + cell_length_2D >= 0) {
117 vel[nbx](i, j, k, 0) = current + local_profile[0];
118 vel[nbx](i, j, k, 1) = local_profile[1];
119 vel[nbx](i, j, k, 2) = local_profile[2];
120 }
121 });
122 amrex::Gpu::synchronize();
123 }

155 const amrex::Real g = wdata.g;

161 const amrex::Real x =
162 amrex::max(problo[0], problo[0] + (i + 0.5) * dx[0]);

164

166 const amrex::Real omega = std::pow(
167 wave_number * g * std::tanh(wave_number * water_depth),

169 const amrex::Real phase =
170 wave_number * (x - current * time) - omega * time -

183 std::cosh(
184 wave_number *

186 std::sinh(wave_number * water_depth) *
187 std::cos(phase) +

189 vel[nbx](i, j, k, 1) = 0.0;
