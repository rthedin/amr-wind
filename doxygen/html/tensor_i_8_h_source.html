217AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE Tensor
218quaternion(const Vector& axis, const amrex::Real angle)
219{
220 const amrex::Real ang = -1.0_rt * utils::radians(angle);
221 const amrex::Real cval = std::cos(0.5_rt * ang);
222 const amrex::Real sval = std::sin(0.5_rt * ang);
223 const auto vmag = mag(axis);
224 const amrex::Real q0 = cval;
225 const amrex::Real q1 = sval * axis.x() / vmag;
226 const amrex::Real q2 = sval * axis.y() / vmag;
227 const amrex::Real q3 = sval * axis.z() / vmag;
228
229 Tensor t;
230 t.xx() = (q0 * q0) + (q1 * q1) - (q2 * q2) - (q3 * q3);
231 t.xy() = 2.0_rt * (q1 * q2 - q0 * q3);
232 t.xz() = 2.0_rt * (q0 * q2 + q1 * q3);
233
234 t.yx() = 2.0_rt * (q1 * q2 + q0 * q3);
235 t.yy() = (q0 * q0) - (q1 * q1) + (q2 * q2) - (q3 * q3);
236 t.yz() = 2.0_rt * (q2 * q3 - q0 * q1);
237
238 t.zx() = 2.0_rt * (q1 * q3 - q0 * q2);
239 t.zy() = 2.0_rt * (q0 * q1 + q2 * q3);
240 t.zz() = (q0 * q0) - (q1 * q1) - (q2 * q2) + (q3 * q3);
241
242 return t;
243}
