55 idx[1];

57 Stencil::c11 * phi(i, j, k, 2) +
58 Stencil::c12 * phi(i, j - 1, k, 2)) *
59 idx[1];

61 Stencil::c21 * phi(i, j, k, 0) +
62 Stencil::c22 * phi(i, j, k - 1, 0)) *
63 idx[2];

65 Stencil::c21 * phi(i, j, k, 1) +
