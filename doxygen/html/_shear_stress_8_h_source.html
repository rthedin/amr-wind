176 AMREX_GPU_DEVICE AMREX_FORCE_INLINE amrex::Real
177 calc_vel_x(amrex::Real u, amrex::Real wspd) const
178 {
179 amrex::Real Cd = 0.001;
180 const amrex::Real c = 7e-5;
181 if (wspd_mean <= 5.0) {
