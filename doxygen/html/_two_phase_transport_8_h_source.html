138 inline std::unique_ptr<ScratchField> mu() override
139 {
140 // Select the interface capturing method
141 auto mu = m_repo.create_scratch_field(1, m_ngrow);
142
143 if (m_ifacetype == InterfaceCapturingMethod::VOF) {
144
145 const auto& vof = m_repo.get_field("vof");
146
147 for (int lev = 0; lev < m_repo.num_active_levels(); ++lev) {
148 for (amrex::MFIter mfi((*mu)(lev)); mfi.isValid(); ++mfi) {
149 const auto& vbx = mfi.growntilebox();
150 const auto& volfrac = vof(lev).const_array(mfi);
151 const auto& visc = (*mu)(lev).array(mfi);
152 const amrex::Real mu1 = m_mu1;
153 const amrex::Real mu2 = m_mu2;
154 amrex::ParallelFor(
155 vbx,
156 [=] AMREX_GPU_DEVICE(int i, int j, int k) noexcept {
157 visc(i, j, k) = mu1 * volfrac(i, j, k) +
158 mu2 * (1.0 - volfrac(i, j, k));
159 });
160 }
161 }
162
163 } else if (m_ifacetype == InterfaceCapturingMethod::LS) {
164
165 const auto& levelset = m_repo.get_field("levelset");
166 const auto& geom = m_repo.mesh().Geom();
167
168 for (int lev = 0; lev < m_repo.num_active_levels(); ++lev) {
169 for (amrex::MFIter mfi((*mu)(lev)); mfi.isValid(); ++mfi) {
170 const auto& vbx = mfi.growntilebox();
171 const auto& dx = geom[lev].CellSizeArray();
172 const auto& visc = (*mu)(lev).array(mfi);
173 const auto& phi = levelset(lev).const_array(mfi);
174 const amrex::Real eps =
175 std::cbrt(2. * dx[0] * dx[1] * dx[2]);
176 const amrex::Real mu1 = m_mu1;
177 const amrex::Real mu2 = m_mu2;
178 amrex::ParallelFor(
179 vbx,
180 [=] AMREX_GPU_DEVICE(int i, int j, int k) noexcept {
181 amrex::Real smooth_heaviside;
182 if (phi(i, j, k) > eps) {
183 smooth_heaviside = 1.0;
184 } else if (phi(i, j, k) < -eps) {
185 smooth_heaviside = 0.;
186 } else {
187 smooth_heaviside =
188 0.5 *
189 (1.0 + phi(i, j, k) / eps +
190 1.0 / M_PI *
191 std::sin(phi(i, j, k) * M_PI / eps));
192 }
193 visc(i, j, k) = mu1 * smooth_heaviside +
194 mu2 * (1.0 - smooth_heaviside);
195 });
196 }
197 }
198 }
199 return mu;
200 }

274 scalar_diffusivity(const std::string& scalar_name) override
275 {
276 amrex::Real lam_schmidt = laminar_schmidt(scalar_name);

278 amrex::Real inv_schmidt = 1.0 / lam_schmidt;

305 inline void beta_impl(
306 const int lev,
307 const amrex::MFIter& mfi,
308 const amrex::Box& bx,
309 const amrex::Array4<amrex::Real>& beta) const override
310 {
311
312 if (m_constant_beta > 0.0) {
313 const amrex::Real beta_val = m_constant_beta;
314 amrex::ParallelFor(
315 bx, [=] AMREX_GPU_DEVICE(int i, int j, int k) noexcept {
316 beta(i, j, k) = beta_val;
317 });
318 } else if (m_repo.field_exists("reference_temperature")) {
319 const auto& temp0 = m_repo.get_field("reference_temperature");
320 const auto& temp0_arr = temp0(lev).const_array(mfi);
321 amrex::ParallelFor(
322 bx, [=] AMREX_GPU_DEVICE(int i, int j, int k) noexcept {
323 beta(i, j, k) = 1.0 / temp0_arr(i, j, k);
324 });
325 } else {
326 const amrex::Real beta_val = 1.0 / m_reference_temperature;
327 amrex::ParallelFor(
328 bx, [=] AMREX_GPU_DEVICE(int i, int j, int k) noexcept {
329 beta(i, j, k) = beta_val;
330 });
331 }
332
333 if (m_ifacetype == InterfaceCapturingMethod::VOF) {
334 const auto& vof = m_repo.get_field("vof");
335 const auto& vof_arr = vof(lev).const_array(mfi);
336 amrex::ParallelFor(
337 bx, [=] AMREX_GPU_DEVICE(int i, int j, int k) noexcept {
338 if (vof_arr(i, j, k) > constants::TIGHT_TOL) {
339 beta(i, j, k) = 0.0;
340 }
341 });
342 } else if (m_ifacetype == InterfaceCapturingMethod::LS) {
343 const auto& levelset = m_repo.get_field("levelset");
344 const auto& dx = m_repo.mesh().Geom(lev).CellSizeArray();
345 const auto& phi_arr = levelset(lev).const_array(mfi);
346 const amrex::Real eps = std::cbrt(2. * dx[0] * dx[1] * dx[2]);
347 amrex::ParallelFor(
348 bx, [=] AMREX_GPU_DEVICE(int i, int j, int k) noexcept {
349 amrex::Real smooth_heaviside;
350 if (phi_arr(i, j, k) > eps) {
351 smooth_heaviside = 1.0;
352 } else if (phi_arr(i, j, k) < -eps) {
353 smooth_heaviside = 0.;
354 } else {
355 smooth_heaviside =
356 0.5 * (1.0 + phi_arr(i, j, k) / eps +
357 1.0 / M_PI *
358 std::sin(phi_arr(i, j, k) * M_PI / eps));
359 }
360 beta(i, j, k) *= (1 - smooth_heaviside);
361 });
362 }
