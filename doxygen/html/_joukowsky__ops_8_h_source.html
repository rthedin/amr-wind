88{
89 void operator()(typename Joukowsky::DataType& data)
90 {
91 ops::base::allocate_basic_grid_quantities<Joukowsky>(data);
92
93 auto& meta = data.meta();
94 meta.tip_correction.resize(meta.num_vel_pts_r);
95 meta.root_correction.resize(meta.num_vel_pts_r);
96 meta.f_normal.resize(meta.num_vel_pts_r);
97 meta.f_theta.resize(meta.num_force_pts);
98
99 std::ranges::fill(meta.tip_correction, 1.0_rt);
100
101 if (meta.use_root_correction) {
102 const amrex::Real dx = 1.0_rt / meta.num_vel_pts_r;
103
104 const auto a = meta.root_correction_coefficient;
105 const auto b = meta.root_correction_exponent;
106
107 const auto delta = amrex::max<amrex::Real>(
108 meta.vortex_core_size / meta.radius(),
109 vs::DTraits<amrex::Real>::eps());
110
111 const auto factor = dx / delta;
112
113 for (int i = 0; i < meta.root_correction.size(); ++i) {
114 meta.root_correction[i] =
115 1.0_rt - std::exp(-a * std::pow((i + 0.5_rt) * factor, b));
116 }
117 } else {
118 std::ranges::fill(meta.root_correction, 1.0_rt);
119 }
120
121 joukowsky::update_disk_points(data);
122 }

140template <>
141struct ComputeForceOp<Joukowsky, ActSrcDisk>
142{
143 void operator()(Joukowsky::DataType& data)
144 {
145 // Equation comments refer to Sorenson 2020 (full reference above)
146 auto& grid = data.grid();
147 auto& ddata = data.meta();
148 const amrex::Real machine_eps = vs::DTraits<amrex::Real>::eps();
149 ddata.disk_force *= 0.0_rt;
150
151 const amrex::Real uInfSqr = base::compute_reference_velocity_sqr(ddata);
152 const amrex::Real U_ref =
153 amrex::max<amrex::Real>(std::sqrt(uInfSqr), machine_eps);
154 base::set_thrust_coefficient(ddata, uInfSqr);
155 joukowsky::set_current_angular_velocity(ddata, uInfSqr);
156
157 const amrex::Real Ct = ddata.current_ct;
158 const amrex::Real dx = ddata.dr / ddata.radius();
159
160 const ops::base::AreaComputer area(
161 ddata.radius(), ddata.num_vel_pts_r, ddata.num_vel_pts_t);
162
163 const amrex::Real lambda =
164 ddata.radius() * ddata.current_angular_velocity / U_ref;
165
166 ddata.current_tip_speed_ratio = lambda;
167
168 const amrex::Real lambda_2 = lambda * lambda;
169
170 // step 0: compute tip correction based on current wind speed (eq 9)
171 if (ddata.use_tip_correction) {
172 auto& f = ddata.tip_correction;
173 for (int i = 0; i < f.size(); ++i) {
174 const auto x = (i + 0.5_rt) * dx;
175 f[i] = 2.0_rt / std::numbers::pi_v<amrex::Real> *
176 std::acos(
177 std::exp(
178 -0.5_rt * ddata.num_blades *
179 std::sqrt(1.0_rt + lambda_2) * (1.0_rt - x)));
180 }
181 }
182
183 // Compute the solid body term S0 (Sorensen 2022, eq. 27)
184 amrex::Real S0 = 0.0_rt;
185 const amrex::Real alpha1 = ddata.S0_alpha1;
186 const amrex::Real alpha2 = ddata.S0_alpha2;
187
188 if (ddata.Ct_rated > 0.0_rt) {
189 if (Ct < ddata.Ct_rated) {
190 S0 = alpha1 * kynema_sgf::utils::powi(
191 (ddata.Ct_rated - Ct) / ddata.Ct_rated, 3);
192 } else {
193 S0 = alpha2 * (ddata.Ct_rated - Ct) / ddata.Ct_rated;
194 }
195 }
196
197 // step 1: compute Ct, a1 and a2 coefficients (eq 16)
198 // a3, a4, a5 coefficients are from Sorensen, 2022
199 amrex::Real a1 = 0.0_rt;
200 amrex::Real a2 = 0.0_rt;
201 amrex::Real a3 = 0.0_rt;
202 amrex::Real a4 = 0.0_rt;
203 amrex::Real a5 = 0.0_rt;
204
205 for (int ip = 0; ip < ddata.num_vel_pts_r; ip++) {
206 const amrex::Real x =
207 amrex::max<amrex::Real>((ip + 0.5_rt) * dx, machine_eps);
208
209 a1 += kynema_sgf::utils::powi(ddata.tip_correction[ip], 2) *
210 kynema_sgf::utils::powi(ddata.root_correction[ip], 2) / x *
211 dx;
212
213 a2 += ddata.tip_correction[ip] * ddata.root_correction[ip] * x * dx;
214
215 a3 += kynema_sgf::utils::powi(ddata.tip_correction[ip], 2) *
216 kynema_sgf::utils::powi(ddata.root_correction[ip], 2) * x *
217 dx;
218
219 a4 += ddata.tip_correction[ip] * ddata.root_correction[ip] *
220 kynema_sgf::utils::powi(x, 3) * dx;
221
222 a5 += kynema_sgf::utils::powi(ddata.tip_correction[ip], 2) *
223 kynema_sgf::utils::powi(ddata.root_correction[ip], 2) *
224 kynema_sgf::utils::powi(x, 3) * dx;
225 }
226
227 // step 2: determine the circulation (q0) from a1, a2 and Ct
228 // (eq. 18, Sorensen, 2022)
229
230 // sqrt[(a2*L+a3*S0)^2+a1*(0.5_rt*Ct-2*a4*L*S0-a5*S0^2)]-(a2*L+a3*S0)
231 // q0 = --------------------------------------------------------------
232 // a1
233
234 const amrex::Real term1 =
235 kynema_sgf::utils::powi((a2 * lambda) + (a3 * S0), 2);
236 const amrex::Real term2 =
237 a1 * (0.5_rt * Ct - 2.0_rt * a4 * lambda * S0 - a5 * S0 * S0);
238 const amrex::Real term3 = (a2 * lambda) + (a3 * S0);
239
240 const amrex::Real numerator = std::sqrt(term1 + term2) - term3;
241
242 const amrex::Real denominator =
243 amrex::max<amrex::Real>(a1, machine_eps);
244
245 const amrex::Real q0 = numerator / denominator;
246
247 // step 3: compute normal force (fz) and azimuthal force (f_theta) (eq
248 // 13) and cp (eq 20)
249
250 VecSlice disk_velocity = ::kynema_sgf::utils::slice(
251 grid.vel, ddata.num_force_pts, ddata.num_force_pts);
252
253 amrex::Real moment = 0.0_rt;
254
255 int ip = 0;
256
257 const auto dTheta = kynema_sgf::utils::two_pi() / ddata.num_vel_pts_t;
258 const auto current_time = data.sim().time().current_time();
259
260 for (int i = 0; i < ddata.num_vel_pts_r; i++) {
261 const amrex::Real& F = ddata.tip_correction[i];
262 const amrex::Real& g = ddata.root_correction[i];
263
264 const amrex::Real x =
265 amrex::max<amrex::Real>((i + 0.5_rt) * dx, machine_eps);
266
267 const amrex::Real f_z =
268 (q0 / x + S0 * x) * g * F *
269 (lambda * x + 0.5_rt * (q0 / x + S0 * x) * g * F);
270
271 ddata.f_normal[i] = f_z;
272
273 const auto dArea = area.area_section(i);
274
275 for (int j = 0; j < ddata.num_vel_pts_t; j++, ip++) {
276 const auto point_current = grid.pos[ip];
277 const auto theta_angle = j * dTheta;
278
279 // TODO add sign convention for rotation (+/- RPM)
280 // Negative sign on theta_vec means turbine is rotating
281 // clockwise when standing upstream looking downstream
282 const auto theta_vec = -utils::compute_tangential_vector(
283 ddata.center, ddata.normal_vec, point_current);
284
285 // normal vec by definition is opposite of the wind direction
286 // so we need to flip the sign to give the actual disk velocity
287 const amrex::Real u_disk_ij =
288 -ddata.normal_vec & disk_velocity[ip];
289
290 const amrex::Real f_theta =
291 u_disk_ij / U_ref * (q0 / x + S0 * x) * g * F;
292 ddata.f_theta[ip] = f_theta;
293
294 // eq 18
295 moment += x * ddata.radius() * f_theta * dArea;
296
297 const amrex::Real f_awc =
298 ddata.awc_amplitude *
299 std::cos(
300 (ddata.awc_angular_frequency * current_time) -
301 (ddata.awc_azimuthal_mode * theta_angle) +
302 ddata.awc_clocking_angle);
303
304 // Kinematic force [m^4/s^2]: omit the density factor so the
305 // disk force enters ActuatorForcing in the same [m/s^2]
306 // convention as every other ICNS source (see uniform_ct_ops
307 // and ActuatorForcing.cpp). current_power / current_cp below
308 // keep their own density factor.
309 grid.force[ip] = ((1 + f_awc) * f_z * ddata.normal_vec +
310 f_theta * theta_vec) *
311 uInfSqr * dArea;
312
313 ddata.disk_force =
314 ddata.disk_force + ddata.density * grid.force[ip];
315 }
316 }
317
318 // equation 20
319 const amrex::Real eq_20_denominator = amrex::max<amrex::Real>(
320 0.5_rt * std::numbers::pi_v<amrex::Real> * ddata.density *
321 kynema_sgf::utils::powi(ddata.radius(), 2) *
322 kynema_sgf::utils::powi(U_ref, 3),
323 machine_eps);
324
325 ddata.current_power =
326 ddata.density * uInfSqr * moment * ddata.current_angular_velocity;
327
328 ddata.current_cp = ddata.density * uInfSqr * moment *
329 ddata.current_angular_velocity / eq_20_denominator;
330#ifdef KYNEMA_SGF_USE_HELICS
331 if (data.info().is_root_proc && data.sim().helics().is_activated()) {
332
333 std::cout << "turbine: " << data.info().id
334 << " jouk power: " << ddata.current_power
335 << " uinfsqr: " << uInfSqr << " moment: " << moment
336 << " ang vel: " << ddata.current_angular_velocity
337 << " normal vec: " << ddata.normal_vec[0] << ' '
338 << ddata.normal_vec[1] << std::endl;
339
340 data.sim().helics().m_turbine_power_to_controller[data.info().id] =
341 ddata.current_power;
342 const amrex::Real turbine_angle = std::atan2(
343 ddata.reference_velocity[1], ddata.reference_velocity[0]);
344 data.sim()
345 .helics()
346 .m_turbine_wind_direction_to_controller[data.info().id] =
347 270.0_rt - kynema_sgf::utils::degrees(turbine_angle);
348 }
349#endif
350 }
351};
